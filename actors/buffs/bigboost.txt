actor BigBoost : CustomInventory 10492
{
  +INVENTORY.ALWAYSPICKUP
  +INVENTORY.UNDROPPABLE
  Inventory.PickupMessage "Energy Boost!!!"
  Inventory.PickupSound "misc/p_pkup"
  +COUNTITEM
  states
  {
  Spawn:
	TNT1 A 0 NoDelay A_PrintBold("BigBoost Spawn")
    AWI3 A -1
    stop
  Pickup:
	TNT1 A 0 NoDelay A_PrintBold("BigBoost Frightened")
   	TNT1 A 0 A_SetArg(0, 30)
    TNT1 A 0 A_ChangeFlag("FRIGHTENED", TRUE)
    TNT1 A 35 A_SetArg(0, args[0]-1)
    TNT1 A 0 A_JumpIf(args[0]==0, "RemoveFlag")

	
    //TNT1 A 0 A_GiveInventory ("Soulsphere", 2)
    //TNT1 A 0 A_GiveInventory ("BFG9000")
    goto Pickup + 3
	RemoveFlag:
	
	TNT1 A 0 NoDelay A_PrintBold("BigBoost No Longer Frightened")
    TNT1 A 0 A_ChangeFlag("FRIGHTENED", FALSE)
	stop
  Death:
  
	TNT1 A 0 NoDelay A_PrintBold("BigBoost Death")
    EMBX C -1
    Stop
  }
}
