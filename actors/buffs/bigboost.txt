actor BigBoost : CustomInventory 10492
{
  +INVENTORY.ALWAYSPICKUP
  +INVENTORY.UNDROPPABLE
  Inventory.PickupMessage "Energy Boost!!!"
  Inventory.PickupSound "misc/p_pkup"
  +COUNTITEM
  states
  {
  Spawn:
	//TNT1 A 0 NoDelay A_PrintBold("BigBoost Spawn")
	TNT1 A 0
    stop
  Pickup:
	TNT1 A 35 A_PrintBold("A")
	TNT1 A 35 A_PrintBold("B")
	TNT1 A 35 A_PrintBold("C")
	TNT1 A 35 A_PrintBold("D")
	TNT1 A 35 A_PrintBold("E")
    TNT1 A 0 A_ChangeFlag("FRIGHTENED", TRUE)
	goto CountdownLoop
	CountdownLoop:
	TNT1 A 120
    TNT1 A 120 A_JumpIfInventory("DecrementCount", 20, "RemoveFlag")
    TNT1 A 120 A_GiveInventory("DecrementCount", 1)
	loop
	RemoveFlag:
	
	TNT1 A 0 A_PrintBold("Remove fear")
    TNT1 A 0 A_ChangeFlag("FRIGHTENED", FALSE)
	stop
  Death:
	//TNT1 A 0 NoDelay A_PrintBold("BigBoost Death")
    TNT1 A 0
    Stop
  }
}
